download_and_read_district <- function(number) {

    # Define the base URL and construct the full URL and destination file name#
    url_base <- "https://cdmaps.polisci.ucla.edu/shp/"
    #combining the url used to extract from the web#
    file_url <- paste0(url_base, "districts", number, ".zip")
    #telling it where to store the Congressional district data output# 
    dest_file <- paste0("districts", number, ".zip")
    
    # Check if the file already exists before downloading#
    if (!file.exists(dest_file)) {
      download.file(file_url, destfile = dest_file)
      cat("Downloaded:", dest_file, "\n")
    } else {
      cat("File already exists:", dest_file, "\n")
    }
}

process_district <- function(district_data) {
  district_data |>
    group_by(STATENAME) |>
    mutate(geometry = st_make_valid(geometry),
           STATENAME = toupper(STATENAME)) |>
    summarize(geometry = st_union(geometry), .groups = "drop")
}
#Creating the function to aggregate all the maps to State Level visualization

District-Wide Winner-Take-All

This provides each state with each District going to the winning party, except for the additional State seats (2 seats) going to the majority presidential candidate in that state.

district_winner <- House_votes |>
  group_by(state, district, year) |>
  # Use slice_max to keep the row with the maximum votes, including party_simplified
  slice_max(order_by = candidatevotes, n = 1, with_ties = FALSE) |>
  # Select necessary columns
  summarize(candidate = first(candidate),
            party = first(party),  # Retain the party of the top candidate
            district_ecvs = 1, .groups = "drop") |>
  group_by(year,party) |>
  summarize(seats_won = sum(district_ecvs, na.rm = TRUE))

state_winner <- House_votes |>
  group_by(state, year, candidate, party) |>
  # Sum votes across all districts to get state-wide totals
  summarize(total_votes = sum(candidatevotes), .groups = "drop") |>

State-Wide Proportional

This method applies the proportion of votes the candidate got across the number of seats to the individual state.

state_proportional <- President_votes |>
  group_by(state, year) |>
  mutate(total_votes = sum(candidatevotes)) |>
  #Calculating number of votes at each candidate received by the number of total votes
  left_join(House_votes |>
              group_by(state,year) |>
              summarize(Num_districts = max(district, na.rm = FALSE)+2)
            ,join_by("state","year")) |>
  mutate(candidate_percentage = candidatevotes / total_votes,
         candidate_ecvs = round(candidate_percentage * Num_districts)) |>
  select(state, year, candidate, candidate_ecvs, party_simplified) |>
  group_by(party_simplified,year) |>
  summarize(seats_won = sum(candidate_ecvs)) |>
  filter(seats_won != 0)

National Proportional

This is pretty obvious as the majority rules logic, with the President getting the most votes winning the majority of seats.

national_proportional <- President_votes |>
  group_by(candidate, year,party_simplified) |>
  summarize(candidate_total_votes = sum(candidatevotes), .groups = "drop") |>
  group_by(year) |>
  mutate(year_total_votes = sum(candidate_total_votes)) |>
  ungroup() |>
  mutate(national_percentage = candidate_total_votes / year_total_votes,
         candidate_ecvs = round(national_percentage * 538)) |>  # 538 total ECVs
  filter(candidate_ecvs != 0) |>
  group_by(year,party_simplified) |>
  summarize(candidate_ecvs = sum(candidate_ecvs))

ggplot(national_proportional, aes(x = year, y = candidate_ecvs, color = party_simplified)) +
